prototype  is mechanism by which js objects inherit features from one another.

let a = {
  name: "mayur",
  year: 1299,
};


const firstPrototype = {
  run: () => {
    alert("run");
  },
};

a.__proto__ = firstPrototype;

const secondprototype = {
  scholl: "maelganon",
};

a.__proto__ = secondprototype;

console.log(a.name);
// console.log(a.run());
console.log(a.scholl);
console.log(a.company)
